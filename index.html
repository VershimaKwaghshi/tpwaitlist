// ========================================== // 1. GLOBAL CONFIGURATION & INITIALIZATION // ========================================== const SUPABASE_URL = "YOUR_SUPABASE_PROJECT_URL"; const SUPABASE_ANON_KEY = "YOUR_SUPABASE_ANON_KEY"; if (typeof window.supabase === 'undefined') { console.warn("Supabase library did not load from CDN. Join Waitlist will show a friendly error instead of crashing."); } const supabase = (SUPABASE_URL.startsWith("YOUR_") || typeof window.supabase === 'undefined') ? null : window.supabase.createClient(SUPABASE_URL, SUPABASE_ANON_KEY); // ========================================== // 2. TWO-DIMENSIONAL CHART ENGINE // ========================================== function initChart2D() { const path = document.getElementById('chartPath'); const dot = document.getElementById('chartDot'); if (!path || !dot) return; try { const points = [ [0, 150], [80, 128], [150, 140], [230, 96], [300, 118], [340, 176], [365, 179], [400, 150], [470, 104], [560, 118], [650, 70], [760, 54] ]; const pathData = points.map((p, i) => (i === 0 ? 'M' : 'L') + p[0] + ',' + p[1]).join(' '); path.setAttribute('d', pathData); let length = path.getTotalLength(); if (length === 0) { length = 900; } path.style.strokeDasharray = length; path.style.strokeDashoffset = length; let startTime = null; function animateChart(ts) { try { if (!startTime) startTime = ts; const progress = Math.min((ts - startTime) / 2600, 1); const eased = 1 - Math.pow(1 - progress, 3); path.style.strokeDashoffset = length * (1 - eased); if (length > 0) { const p = path.getPointAtLength(length * eased); if (p) { dot.setAttribute('cx', p.x); dot.setAttribute('cy', p.y); } } if (progress < 1) { requestAnimationFrame(animateChart); } } catch (frameErr) { console.error("Asynchronous error caught inside 2D loop:", frameErr); } } requestAnimationFrame(animateChart); } catch (err) { console.error("2D chart failed to initialize:", err); } } // ========================================== // 3. THREE-DIMENSIONAL MARKET PULSE ENGINE // ========================================== function initMarket3D() { const container = document.getElementById('chart3DStage'); if (!container || typeof THREE === 'undefined') { console.warn("Three.js not available, skipping 3D chart."); return; } try { const scene = new THREE.Scene(); let width = container.clientWidth || 760; let height = container.clientHeight || 280; const camera = new THREE.PerspectiveCamera(45, width / height, 0.1, 100); camera.position.set(0, 5.5, 13); camera.lookAt(0, 0, 0); const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true }); renderer.setSize(width, height); renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2)); container.appendChild(renderer.domElement); scene.add(new THREE.AmbientLight(0xffffff, 0.4)); const dirLight = new THREE.DirectionalLight(0xC7A252, 1.2); dirLight.position.set(6, 10, 8); scene.add(dirLight); const fillLight = new THREE.DirectionalLight(0xffffff, 0.25); fillLight.position.set(-6, 4, -6); scene.add(fillLight); const barCount = 22; const spacing = 0.8; const startX = -((barCount - 1) * spacing) / 2; const bars = []; for (let i = 0; i < barCount; i++) { const geo = new THREE.BoxGeometry(0.46, 1, 0.46); const warmth = i / barCount; const color = new THREE.Color().setHSL(0.11, 0.55, 0.32 + warmth * 0.28); const mat = new THREE.MeshStandardMaterial({ color, metalness: 0.35, roughness: 0.45 }); const bar = new THREE.Mesh(geo, mat); bar.position.x = startX + i * spacing; scene.add(bar); bars.push({ mesh: bar, phase: Math.random() * Math.PI * 2, speed: 0.5 + Math.random() * 0.7 }); } function resize3D() { const w = container.clientWidth; const h = container.clientHeight; if (w === 0 || h === 0) return; camera.aspect = w / h; camera.updateProjectionMatrix(); renderer.setSize(w, h); } window.addEventListener('resize', resize3D); let t = 0; function animate3D() { try { requestAnimationFrame(animate3D); t += 0.016; bars.forEach((b) => { const dynamicHeight = 1.1 + Math.sin(t * b.speed + b.phase) * 1.3 + 0.6; b.mesh.scale.y = Math.max(0.25, dynamicHeight); b.mesh.position.y = (b.mesh.scale.y) / 2; }); scene.rotation.y = Math.sin(t * 0.15) * 0.35; renderer.render(scene, camera); } catch (animErr) { console.error("Asynchronous error caught inside 3D loop:", animErr); } } animate3D(); } catch (err) { console.error("3D market pulse failed to initialize:", err); } } // ========================================== // 4. LIFECYCLE INITIALIZATION BINDING // ========================================== window.addEventListener('load', () => { initChart2D(); initMarket3D(); }); // ========================================== // 5. REGISTRATION AND WAITLIST LOGIC // ========================================== const urlParams = new URLSearchParams(window.location.search); const refFromUrl = urlParams.get('ref'); if (refFromUrl) { const refInput = document.getElementById('referralCode'); if (refInput) refInput.value = refFromUrl; } let turnstileToken = null; function onTurnstileVerified(token) { turnstileToken = token; } function generateReferralCode() { const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789'; let code = 'TP-'; for (let i = 0; i < 6; i++) { code += chars.charAt(Math.floor(Math.random() * chars.length)); } return code; } let pendingSignup = null; const stepForm = document.getElementById('stepForm'); const stepOtp = document.getElementById('stepOtp'); const dashboard = document.getElementById('dashboard'); const step1Error = document.getElementById('step1Error'); const otpError = document.getElementById('otpError'); function showError(el, msg) { if (!el) return; el.textContent = msg; el.classList.remove('hidden'); } function clearError(el) { if (!el) return; el.textContent = ''; el.classList.add('hidden'); } if (stepForm) { stepForm.addEventListener('submit', async function(e) { e.preventDefault(); clearError(step1Error); const fullName = document.getElementById('fullName')?.value.trim(); const email = document.getElementById('email')?.value.trim().toLowerCase(); const phone = document.getElementById('phone')?.value.trim(); const country = document.getElementById('country')?.value; const role = document.querySelector('input[name="role"]:checked')?.value; const referredBy = document.getElementById('referralCode')?.value.trim() || null; if (!fullName || !email || !country) { showError(step1Error, "Please fill in your name, email, and country."); return; } const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; if (!emailPattern.test(email)) { showError(step1Error, "That email doesn't look right."); return; } if (!supabase) { showError(step1Error, "Backend not connected yet. Add your Supabase credentials at the top of this file."); return; } const btn = document.getElementById('submitStep1'); if (btn) { btn.disabled = true; btn.textContent = 'Sending code...'; } pendingSignup = { fullName, email, phone, country, role, referredBy }; const { error } = await supabase.auth.signInWithOtp({ email, options: { shouldCreateUser: true } }); if (btn) { btn.disabled = false; btn.textContent = 'Continue'; } if (error) { showError(step1Error, error.message); return; } const hint = document.getElementById('otpHint'); if (hint) hint.textContent = "Enter the 6 digit code sent to " + email; stepForm.classList.add('hidden'); if (stepOtp) stepOtp.classList.remove('hidden'); }); } if (stepOtp) { stepOtp.addEventListener('submit', async function(e) { e.preventDefault(); clearError(otpError); if (!pendingSignup) return; if (!supabase) { showError(otpError, "Backend not connected yet. Add your Supabase credentials at the top of this file."); return; } const code = document.getElementById('otpCode')?.value.trim(); const btn = document.getElementById('submitOtp'); if (btn) { btn.disabled = true; btn.textContent = 'Verifying...'; } const { data, error } = await supabase.auth.verifyOtp({ email: pendingSignup.email, token: code, type: 'email' }); if (btn) { btn.disabled = false; btn.textContent = 'Verify'; } if (error) { showError(otpError, error.message); return; } const userId = data.user.id; const constRefCode = generateReferralCode(); try { const { error: upsertError } = await supabase.from('users').upsert({ id: userId, full_name: pendingSignup.fullName, email: pendingSignup.email, phone: pendingSignup.phone, country: pendingSignup.country, role: pendingSignup.role, referral_code: constRefCode, referred_by: pendingSignup.referredBy || null, status: 'pending_activation', email_verified: true }); if (upsertError) { showError(otpError, upsertError.message); return; } await loadDashboard(userId, constRefCode); } catch (err) { console.error("Database connection failure:", err); } }); } const resendBtn = document.getElementById('resendOtp'); if (resendBtn) { resendBtn.addEventListener('click', async function() { if (!pendingSignup || !supabase) { showError(otpError, "Backend not connected yet."); return; } await supabase.auth.signInWithOtp({ email: pendingSignup.email, options: { shouldCreateUser: true } }); const hint = document.getElementById('otpHint'); if (hint) hint.textContent = 'Code resent successfully.'; }); } // ========================================== // 6.
